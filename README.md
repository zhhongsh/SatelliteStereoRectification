```

  author    = {Nurollah Tatar and
               Hossein Arefi},
  title     = {{Stereo rectification of pushbroom satellite images by robustly estimating the fundamental matrix}},
  journal   = {{International Journal of Remote Sensing (IJRS)}},
  year      = {2019},
  volume    = {40},
  number    = {23},
  pages     = {8879--8898},

}
```
